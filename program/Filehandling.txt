import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileWriter;
import java.io.IOException;
import java.util.Scanner;

public class cwh_111_filejava {
    public static void main(String[] args) {
        File myFile = new File("cwh111file.txt");
        try {
            if (myFile.createNewFile()) {
                System.out.println("File created: " + myFile.getName());
            } else {
                System.out.println("File already exists.");
            }
        } catch (IOException e) {
            System.out.println("Unable to create file");
            e.printStackTrace();
        }
        try{
            FileWriter fileWriter= new FileWriter( "cwh111file.txt");
            fileWriter.write("this firt file abhay");
            fileWriter.close();

        }catch (IOException e){
            e.printStackTrace();
        }


        try{
            File myyFile = new File("cwh111file.txt");
            Scanner sc =new Scanner(myyFile);
            while (sc.hasNextLine()){
                String line =sc.nextLine();
                System.out.println(line);
            }
            sc.close();
        }
        catch (FileNotFoundException e){
            System.out.println("File not found");
            e.printStackTrace();
        }


    }
}
